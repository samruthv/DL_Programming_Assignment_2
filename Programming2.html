  2. Apply the function to y_train and y_test.

def to_one_hot(y, num_class=10):
    matrix=[]
    for i in range(len(y)):
        vec=[0]*num_class
        vec[int(y[i])]=1
        #print(x)
        matrix.append(vec)
    matrix=numpy.array(matrix)
    return(matrix)
